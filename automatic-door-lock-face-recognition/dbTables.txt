CREATE TABLE face_records (
  id SERIAL PRIMARY KEY,
  name VARCHAR(200) NOT NULL,
  image_path TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT now()
);

drop table if exists personnel_information;
CREATE TABLE personnel_information (
	personnel_information_id SERIAL PRIMARY KEY,
	first_name varchar(255),
	middle_name varchar(255),
	last_name  varchar(255),
	email_address varchar(255), 
	phone_number varchar(255)
);

drop table if exists personnel_face_records;
CREATE TABLE personnel_face_records (
	personnel_face_records_id SERIAL PRIMARY KEY,
	personnel_information_id BIGINT REFERENCES personnel_information(personnel_information_id),
	personnel_name VARCHAR(200) NOT NULL,
	image_path TEXT NOT NULL,
	created_at TIMESTAMPTZ DEFAULT now()
);

drop table if exists personnel_door_logs;
CREATE TABLE personnel_door_logs (
	personnel_door_logs_id SERIAL PRIMARY KEY,
	personnel_information_id BIGINT REFERENCES personnel_information(personnel_information_id),
	log_type varchar(255),
	time_log TIMESTAMPTZ DEFAULT now()
);

drop table if exists document_information;
CREATE TABLE document_information (
	document_information_id SERIAL PRIMARY KEY,
	document_rfid_tag varchar(255),
	document_type varchar(255)
);

drop table if exists document_information_logs;
CREATE TABLE document_information_logs(
	document_information_logs_id SERIAL PRIMARY KEY,
	document_information_id BIGINT REFERENCES document_information(document_information_id),
	log_type varchar(255),
	time_log TIMESTAMPTZ DEFAULT now()
);
